    for j in range(3):
        file = testFileName[i*3+j]
        df_part = df[df['']==file]
        group = []
        for col in colnames:
            group.append(list(df_part[df_part['']==col]['gap']))
        ax = plt.subplot(position[j])
        plt.boxplot(group)
        ax.set_xticklabels(colnames, rotation = 90, fontsize=15,color='r')
        ax.set_ylabel('gap %',fontsize=15, color='r')
        ax.set_title(file ,fontsize=20)
    plt.show()
