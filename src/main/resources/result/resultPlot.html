colnames = list(df['算子名称'].drop_duplicates())
colnames.sort()
position = [131,132,133]

for i in range(3):
    plt.figure(i, figsize=[18, 6]) # 子图 
    for j in range(3):
        file = testFileName[i*3+j]
        df_part = df[df['文件名称']==file]
        group = []
        for col in colnames:
            group.append(list(df_part[df_part['算子名称']==col]['gap']))
        ax = plt.subplot(position[j])
        plt.boxplot(group)
        ax.set_xticklabels(colnames, rotation = 90, fontsize=15,color='r')
        ax.set_ylabel('gap %',fontsize=15, color='r')
        ax.set_title(file ,fontsize=20)
    plt.show()
